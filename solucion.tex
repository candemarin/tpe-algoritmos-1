\section{Problemas}
\subsection{Ejercicio 1}

\aux{minasAdyacentes}{t: \tablero, p: \pos}{$\mathbb{Z}$}{ -1 + \sum_{i=0}^{|t|-1} \sum_{j=0}^{|t[i]|-1} \IfThenElse{\text{sonAdyacentes}(t, p, (i,j)) \land t[i][j]}{1}{0}}

\pred{sonAdyacentes}{t: \tablero, p: \pos, q: \pos}{
	\text{/* si p = q el predicado es true */} \\ \\
	\text{posValida(t, p)} \yLuego \\ 
	\text{posValida(t, q)} \yLuego \\ 
	((p[0] - q[0])^2 \leq 1 \land (p[1] - q[1])^2 \leq 1 )
}

\subsection{Ejercicio 2}
\pred{juegoValido}{t: \tablero, j: \jugadas}{
(\forall i : \ent) (0 \leq i < |t| \implica	0 \leq j[i][0][0] < |t[0]| \y 0 \leq j[i][0][1] < |t| \yLuego (minasAdyacentes(t, j[i][0]) = j[i][1])
}
